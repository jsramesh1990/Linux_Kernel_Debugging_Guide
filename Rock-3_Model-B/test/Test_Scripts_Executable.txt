Make All Test Scripts Executable
bash

chmod +x test/integration/*.sh

Run Tests
bash

# Run individual test suite
cd test/integration
./test-system.sh
./test-performance.sh
./test-peripheral.sh

# Run all tests
./run-all-tests.sh

# Or using make
make test-all

# Run specific test with make
make test-system
make test-performance
make test-peripheral

Test Directory Structure
text
